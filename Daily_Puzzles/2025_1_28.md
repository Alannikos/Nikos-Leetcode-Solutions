class Solution {
public:
    vector<int> getRow(int rowIndex) {
        vector<int> ans={1};
        while(rowIndex-->0)
        {
            vector<int> t(ans.size()+1,1);
            for(int i=1;i<t.size()-1;i++)
            {
                t[i]=ans[i]+ans[i-1];
            }
            ans=move(t);
        }
        return ans;
    }
};

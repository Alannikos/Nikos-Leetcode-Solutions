    int jump(vector<int>& nums)
    {
        int ans = 0;  // 记录跳跃次数
        int cur_distance = 0;
        int max_distance = 0;

        for (int i = 0; i < nums.size() - 1; i ++)
        {
            max_distance = max(max_distance, i + nums[i]);
            if (i == cur_distance)
            {
                cur_distance = max_distance;
                ans ++;
            }
        }

        return ans;
    }
};

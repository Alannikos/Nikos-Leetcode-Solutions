class Solution {
public:
    long long minimumMoney(vector<vector<int>>& transactions) {
        int n = transactions.size();
        long long totalloss = 0;
        long long res = 0;

        for (int i = 0; i < n; i ++)
        {
            int cost = transactions[i][0];
            int cashback = transactions[i][1];

            if (cost > cashback)
            {
                totalloss += (cost - cashback);
            }
        }

        for (int i = 0; i < n; i ++)
        {
            int cost = transactions[i][0];
            int cashback = transactions[i][1];

            if (cost <= cashback)
            {
                res = max(res, totalloss + cost);
            }
            else
            {
                res = max(res, totalloss + cashback);
            }
        }

        return res;
    }
};
